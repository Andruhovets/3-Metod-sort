    class BubleSortArray                                         //Метод бульбашки
    {
        public void BubbleSort()
        {
            int i, j;
            int temp;
            Random r = new Random();
            int size = 20;
            int[] mas = new int[size];
                for (i = 0; i < mas.Length; i++)
                {
                    mas[i] = r.Next(0, 101);              // масив заповнюється випадковими числами від 0 до 100 
                }

                Console.WriteLine("Невідсортований масив :");   //Виводить невідсортований масив чисел
                for (i = 0; i < mas.Length; i++)
                {
                    Console.Write(mas[i]+" ");
                }
                  
            
            for (i = 0; i < size - 1; i++)
            {
                for (j = i + 1; j < size; j++)
                {
                    if (mas[i] > mas[j])
                    {
                         temp = mas[i];
                        mas[i] = mas[j];
                        mas[j] = temp;
                    }
                }
            }
            Console.WriteLine();
            Console.WriteLine("Відсортований масив :");     //Виводить відсортований масив чисел
            for (i = 0; i < mas.Length; i++)
            {
                Console.Write(mas[i] + " ");
            }
                  
        }
    }
class InsertionArray                                          //метод включення
{
      public void InsertionSort()
            {
                int i;
            Random r = new Random();
            int size = 20;
            int[] mas = new int[size];
                for (i = 0; i < mas.Length; i++)
                {
                    mas[i] = r.Next(0, 100);           // масив заповнюється випадковими числами від 0 до 100 
                }
                Console.WriteLine();
                Console.WriteLine("Невідсортований масив :"); //Виводить невідсортований масив чисел
                for (i = 0; i < mas.Length; i++)
                {
                    Console.Write(mas[i]+" ");
                }
                int a, b;
                for ( i = 1; i <  mas.Length; i++)
                {
                    a = mas[i];
                    b = i - 1;
                    while (b >= 0 && mas[b] > a)
                    {
                        mas[b + 1] = mas[b];
                        b = b - 1;
                    }
                    mas[b + 1] = a;
                }
                Console.WriteLine();
                Console.WriteLine("Відсортований масив :");    //Виводить відсортований масив чисел
                for (i = 0; i < mas.Length; i++)
                {
                    Console.Write(mas[i] + " ");
                }
                  
    }
}
class Transposition                                            //метод перестановки
{

    public void Transpositin()
    {

        int i;
        Random r = new Random();
        int size = 20;
        int[] mas = new int[size];
        for (i = 0; i < mas.Length; i++)
        {
            mas[i] = r.Next(1, 99);                      // масив заповнюється випадковими числами від 0 до 100 
        }
        Console.WriteLine();
        Console.WriteLine("Невідсортований масив :");         //Виводить невідсортований масив чисел
        for (i = 0; i < mas.Length; i++)
        {
            Console.Write(mas[i] + " ");
        }
        for (i= 1; i < mas.Length; i++)
        {
            int cur = mas[i];
            int j = i;
            while (j > 0 && cur < mas[j - 1])
            {
                mas[j] = mas[j - 1];
                j--;
            }
            mas[j] = cur;
        }
        Console.WriteLine();
        Console.WriteLine("Відсортований масив :");           //Виводить відсортований масив чисел
        for (i = 0; i < mas.Length; i++)
        {
            Console.Write(mas[i] + " ");
        }
    }
